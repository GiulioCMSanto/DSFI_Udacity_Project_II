    """
    Calcula o nmero de Amostras e de Colunas de um Dataframe de nome "name"
    Parameters:
    df: DataFrame a ser analisado
    name: nome do DataFrame

    print("Nmero de Amostras e Colunas em {}: {} Amostras e {} Colunas".format(name,df.shape[0],df.shape[1]))

    """
    Calcula o nmero de Linhas Duplicatadas de um Dataframe de nome "name"
    Parameters:
    df: DataFrame a ser analisado
    name: nome do DataFrame
    """
    print("Nmero de Linhas repetidas em {}: {}".format(name,sum(df.duplicated())))

def pclass_analysis(df):
    """
    Realiza a Anlise das Classes econmicas. Verifica a existncia de Correlaes.
    Apresenta grfico de barras comparando a taxa de sobrevivncia por classe econmica.
    Parameters:
        df: DataFrame a ser analisado
    """
    #Definindo grupo de classes econmicas
    classes = df.groupby(['pclass','survived']).count()['sex']
    
    #Definindo DataFrames Sobreviventes e No Sobreviventes
    survived = df.query("survived == '1'")
    not_survived = df.query("survived == '0'")
    
    #Definindo Nmero total de pessoas a bordo e a proporo por classe
    total_aboard = df['sex'].count()
    proportion = classes/total_aboard

    #Criando lista com proporo de sobreviventes e de no sobreviventes para cada uma das classes (1a, 2a, 3a)
    prop_survived = [proportion.values[1], proportion.values[3], proportion.values[5]]
    prop_not_survived = [proportion.values[0],proportion.values[2],proportion.values[4]]
    
    #Criando ndice Para o Grfico de Barras
    ind = list(proportion.index.levels[0])
    ind = np.arange(len(ind))
    
    #Criando Grfico de Barras Para Comparar Proporo de sobreviventes por classe econmica
    width = 0.35
    plt.bar(ind,prop_survived,width,color='r',alpha=.7,label='Proporo Sobreviventes')
    plt.bar(ind+width,prop_not_survived,width,color='w',alpha=.7,label='Proporo No Sobreviventes')
    locations = ind+width/2
    labels = ['Primeira Classe', 'Segunda Classe', 'Terceira Classe']
    plt.xticks(locations,labels);
    plt.title("Proporo de Sobrevivncia Por Classe Econmica")
    plt.xlabel("Classe Econmica")
    plt.ylabel("Proporo de Sobrevivncia")
    plt.legend();
    
    #Criando Segundo Grfico de Barras com Comparao mais Genrica e Agrupada
    plt.figure()
    sns.barplot(x="sex", y="survived", hue="pclass", data=df);
    plt.xticks([0,1],['Masculino','Feminino'])
    plt.title("Sobreviventes Por Sexo e Classe Econmica")
    plt.xlabel("Sexo")
    plt.ylabel("Sobreviveu?")
    
    
    #Descrio das Classes dos Sobreviventes
    print("Descrio da Classe dos Sobreviventes: \n")
    display(survived['pclass'].describe())

    #Descrio das Classes dos No Sobreviventes
    print("Descrio da Classe dos No Sobreviventes: \n")
    display(not_survived['pclass'].describe())

4.1 Limitações¶

Alguns fatores importantes ao se levantar uma amostra devem ser considerados (Neto, C.):

 - Garantir que a amostra seja representativa da populao, ou seja, a amostra deve ter as mesmas caractersticas bsicas da populao no que diz respeito s variveis que esto sendo pesquisadas.
 -  preciso garantir que as amostras sejam obtidas por processos adequados. Erros no momento da seleo das amostras comprometem todo o trabalho analtico.
 - Deve-se escolher o mtodo de amostragem mais adequado a cada caso (Amostragem Probabilsitca ou No Probabilistica).

Em nosso caso de estudo, estamos trabalhando com dados que nos foram previamente fornecidos. Sendo assim, evidentemente estamos sujeitos a estar trabalhando com dados com vcios de amostragem e que, portanto, no sejam representativos o suficiente da populao.

Alm disso, importante considerar que quanto maior o tamanho da amostra, melhor ser a estimao dos parmetros bem como a capacidade de tirar concluses analticas. Em nosso caso especificamente, como vimos na etapa de Data Wrangle, o nmero de linhas repetidas era bastante grande e, particularmente para o atributo "Idade", o nmero de dados ausentes era enorme. A perda de dados sem dvida um dos fatores que afeta enormemente a anlise, dificultado formular concluses slidas.

Particularmente neste projeto, essas limitaes foram tratadas segregando-se o DataFrame em dois subconjuntos:

- O primeiro eliminou a coluna "Idade", pois esta possui muitos dados ausentes. Desta forma, conseguiu-se otimizar o tamanho da amostra para analisar as demais caractersticas.
- O segundo foi composto por todos os dados aps a retirada de todos os valores nulos. Esse DataFrame teve um conjunto de elementos reduzido mas que permitiu realizar a anlise do atributo "Idade". Evidentemente concluses mais fortes poderiam ter sido tiradas caso tivessemos um melhor grupo de dados.

Vale ressaltar, por fim, que a eliminao dos atributos que no foram analisados tambm contribuiu para otimizar o tamanho da amostra e chegar em concluses satisfatrias.
