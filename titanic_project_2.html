def num_row_col(df,name):
    """
    Calcula o número de Amostras e de Colunas de um Dataframe de nome "name"
    Parameters:
    df: DataFrame a ser analisado
    name: nome do DataFrame
    """
    print("Número de Amostras e Colunas em {}: {} Amostras e {} Colunas".format(name,df.shape[0],df.shape[1]))

def num_repeated_rows(df,name):
    """
    Calcula o número de Linhas Duplicatadas de um Dataframe de nome "name"
    Parameters:
    df: DataFrame a ser analisado
    name: nome do DataFrame
    """
    print("Número de Linhas repetidas em {}: {}".format(name,sum(df.duplicated())))
    
def num_null_col(df,name):
    """
    Calcula que atributos tem valores nulos em um Dataframe de nome "name"
    Parameters:
    df: DataFrame a ser analisado
    name: nome do DataFrame
    """
    print('Atributos com Valores Ausentes em {}: {}'.format(name,df.columns[df.isnull().any()].values))

def age_analysis(df):  
    """
    Realiza a Análise da Idade. Verifica a existência de Correlação entre idade e sobreviventes. Apresenta
    gráfico de correlação de distribuição
    Parameters:
        df: DataFrame a ser analisado
    """
    
    #Plot de Gráfico de Correlação Entre Idade e Sobrevivência
    plt.scatter(df['age'],df['survived'],color='orange');
    plt.title('Sobreviventes x Idade');
    plt.xlabel('Idade');
    plt.ylabel('Sobreviveu?');

    #Criando Um DataFrame para os sobreviventes
    survived = df.query('survived == "1"')

    #Criando um DataFrame para os não sobreviventes
    not_survived = df.query('survived == "0"')

    #Verificando Idade Média dos Sobreviventes
    print("A idade média dos Sobreviventes foi de {} anos".format(survived['age'].mean()))
    print("A idade média dos Não Sobreviventes foi de {} anos".format(not_survived['age'].mean()))

    #Traçando Distribuição para a idade dos sobreviventes
    #survived['age'].plot(kind='hist', figsize=(4,4), title='Distribuição da Idade dos Sobreviventes');
    plt.figure()
    sns.kdeplot(survived['age'], shade=True, legend = False);
    plt.xlabel('Idade');
    plt.title('Distribuição da Variável Idade Entre os Sobreviventes')
    
    #Traçando Distribuição para a idade dos não sobreviventes
    #not_survived['age'].plot(kind='hist', figsize=(4,4), title='Distribuição da Idade dos Não Sobreviventes');
    plt.figure()
    ax = sns.kdeplot(not_survived['age'], shade=True, legend = False);
    plt.xlabel('Idade');
    plt.title('Distribuição da Variável Idade Entre os Não Sobreviventes')

def pclass_analysis(df):
    """
    Realiza a Análise das Classes econômicas. Verifica a existência de Correlações.
    Apresenta gráfico de barras comparando a taxa de sobrevivência por classe econômica.
    Parameters:
        df: DataFrame a ser analisado
    """
    #Definindo grupo de classes econômicas
    classes = df.groupby(['pclass','survived']).count()['sex']
    
    #Definindo DataFrames Sobreviventes e Não Sobreviventes
    survived = df.query("survived == '1'")
    not_survived = df.query("survived == '0'")
    
    #Definindo Número total de pessoas a bordo e a proporção por classe
    total_aboard = df['sex'].count()
    proportion = classes/total_aboard

    #Criando lista com proporção de sobreviventes e de não sobreviventes para cada uma das classes (1a, 2a, 3a)
    prop_survived = [proportion.values[1], proportion.values[3], proportion.values[5]]
    prop_not_survived = [proportion.values[0],proportion.values[2],proportion.values[4]]
    
    #Criando Índice Para o Gráfico de Barras
    ind = list(proportion.index.levels[0])
    ind = np.arange(len(ind))
    
    #Criando Gráfico de Barras Para Comparar Proporção de sobreviventes por classe econômica
    width = 0.35
    plt.bar(ind,prop_survived,width,color='r',alpha=.7,label='Proporção Sobreviventes')
    plt.bar(ind+width,prop_not_survived,width,color='w',alpha=.7,label='Proporção Não Sobreviventes')
    locations = ind+width/2
    labels = ['Primeira Classe', 'Segunda Classe', 'Terceira Classe']
    plt.xticks(locations,labels);
    plt.title("Proporção de Sobrevivência Por Classe Econômica")
    plt.xlabel("Classe Econômica")
    plt.ylabel("Proporção de Sobrevivência")
    plt.legend();
    
    #Criando Segundo Gráfico de Barras com Comparação mais Genérica e Agrupada
    plt.figure()
    sns.barplot(x="sex", y="survived", hue="pclass", data=df);
    plt.xticks([0,1],['Masculino','Feminino'])
    plt.title("Sobreviventes Por Sexo e Classe Econômica")
    plt.xlabel("Sexo")
    plt.ylabel("Sobreviveu?")
    
    
    #Descrição das Classes dos Sobreviventes
    print("Descrição da Classe dos Sobreviventes: \n")
    display(survived['pclass'].describe())

    #Descrição das Classes dos Não Sobreviventes
    print("Descrição da Classe dos Não Sobreviventes: \n")
    display(not_survived['pclass'].describe())

A análise acima mostra que a média de idade das pessoas que sobreviveram é próxima à média de idade das pessoas que não sobreviveram ao acidente. Entretanto, parece existir uma tendência de aumento de idade entre os não sobreviventes. Os diagramas de dispersão mostram, por exemplo, que enquanto a Idade dos Sobreviventes possui uma distribuição mais próxima de uma normal, com média em torno de 28 anos, a Idade dos Não Sobreviventes possui uma distorção forte à direita que justamente corresponde a idades mais elevadas.

Analogamente, se analisarmos o "point plot" acima verificamos graficamente as médias de idade dos sobreviventes e dos não sobreviventes, em que a idade média dos não sobreviventes é um pouco maior.

O gráfico de correlação também sugere um deslocamento para a direita dos pontos que representam os não sobreviventes. Entretanto, como a diferença na média das distruibições de sobreviventes e não sobreviventes é pequena, não podemos concluir de forma definitiva que de fato pessoas mais velhas foram mais afetadas pelo acidente. Algum método de estatística inferencial deve ser utilizado (como um teste de hipótese) para chegarmos em conclusões mais precisas.

Etapa 4: Conclusões¶

Primeiramente, é fundamental mencionar que como estamos apenas no escopo da Estatística Descritiva, não podemos realizar afirmações acertivas sobre nossas conclusões. Entretanto, conseguimos obter bons "insights" que poderiam ser validados posteriormente com ferramentes mais poderosas de análise.

Vamos analisar as conclusões tiradas individualmente para cada pergunta:

1) Pode-se afirmar que o gráfico de correlação sugere uma leve tendência de correlação positiva entre idade e número de mortos. Além disso, os demais gráficos apresentados sustentam essa afirmação: nitidamente o gráfico de dispersão da idade dos sobreviventes possui média inferior à do gráfico de dispersão da idade dos não sobreviventes. Além disso, nota-se que a dispersão da idade dos não sobreviventes é distorcida à direita, justamente a direção que representa aumento da idade. Vale ressaltar, entretanto, que não podemos afirmar de forma sólida que esta correlação positiva entre idade e número de sobreviventes de fato exista. A principal razão disso é o fato de, como explicado na etapa de Data Wrangle, o atributo "Idade" possuir muitos dados ausentes. Sendo assim, a amostra para esta característica possui poucos elementos, dificultando uma conclusão mais sólida.

2) Embora a tendência seja imaginar que pessoas com maior número de crianças ou casais foram mais mais afetados pelo acidente, os gráficos apresentados neste item não sustentam essa afirmação. Nitidamente, a média de crianças e casais entre os sobrevivente e os não sobreviventes é muito parecida. Além disso, os gráficos de contagem não mostram uma evolução do número de crianças e casais entre os que não sobreviveram. Pelo contrário, esse número se comporta de forma semelhante entre os que sobreviveram e os que não sobreviveram, sugerindo que não haja correlação. O gráfico de correlação, por sua vez, pode parecer indicar que exista sim correlação. Entretanto, sua análise não permite uma conclusão forte sem algum elemento inferencial que a reforce.

3) Podemos dizer que James Cameron foi justo em seu filme! De fato, todas as análises neste item sugerem que pessoas que estavam em classe econômica mais baixas (sendo a classe 3 a mais baixa) foram mais prejudicadas, ou seja, estavam em maior proporção entre os que morreram. Além disso, o segundo gráfico mostrado nesse item confirma essa tendência mesmo segregando as pessoas por gênero.

4) Conforme esperávamos, a classe econômica e o preço das passagens estão correlacionados, ou seja, pessoas em classes econômicas melhores dentro do navil (1 e 2) pagaram mais pela passagem. Consequentemente, o resultado deste item foi igual ao do item anterior: as pessoas com passagens mais baratas estavam em maior proporção entre os não sobreviventes e pessoas que pagaram mais pelas suas passagens estava em proporção maior entre os sobreviventes.

4.1 Limitações¶

Alguns fatores importantes ao se levantar uma amostra devem ser considerados (Neto, C.):

 - Garantir que a amostra seja representativa da população, ou seja, a amostra deve ter as mesmas características básicas da população no que diz respeito às variáveis que estão sendo pesquisadas.
 - É preciso garantir que as amostras sejam obtidas por processos adequados. Erros no momento da seleção das amostras comprometem todo o trabalho analítico.
 - Deve-se escolher o método de amostragem mais adequado a cada caso (Amostragem Probabilísitca ou Não Probabilistica).

Em nosso caso de estudo, estamos trabalhando com dados que nos foram previamente fornecidos. Sendo assim, evidentemente estamos sujeitos a estar trabalhando com dados com vícios de amostragem e que, portanto, não sejam representativos o suficiente da população.

Além disso, é importante considerar que quanto maior o tamanho da amostra, melhor será a estimação dos parâmetros bem como a capacidade de tirar conclusões analíticas. Em nosso caso especificamente, como vimos na etapa de Data Wrangle, o número de linhas repetidas era bastante grande e, particularmente para o atributo "Idade", o número de dados ausentes era enorme. A perda de dados é sem dúvida um dos fatores que afeta enormemente a análise, dificultado formular conclusões sólidas.

Particularmente neste projeto, essas limitações foram tratadas segregando-se o DataFrame em dois subconjuntos:

- O primeiro eliminou a coluna "Idade", pois esta possui muitos dados ausentes. Desta forma, conseguiu-se otimizar o tamanho da amostra para analisar as demais características.
- O segundo foi composto por todos os dados após a retirada de todos os valores nulos. Esse DataFrame teve um conjunto de elementos reduzido mas que permitiu realizar a análise do atributo "Idade". Evidentemente conclusões mais fortes poderiam ter sido tiradas caso tivessemos um melhor grupo de dados.

Vale ressaltar, por fim, que a eliminação dos atributos que não foram analisados também contribuiu para otimizar o tamanho da amostra e chegar em conclusões satisfatórias.
